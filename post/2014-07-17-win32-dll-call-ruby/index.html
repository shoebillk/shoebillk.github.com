require 'fiddle/types'

module Win32Test
  extend Fiddle::Importer
  dlload 'user32.dll'

  include Fiddle::Win32Types

  extern "DWORD MessageBoxA(HWND, LPCSTR, LPCSTR, DWORD)"

  def self.msgbox
    MessageBoxA(0, "OK?", "Please Confirm", 0x40)
  end

end
